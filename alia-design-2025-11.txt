

-- ALIA COMPONENTS --

os/window:
 - backed by win32 (d3d9), x11, etc
 - mirrors allegro

os/dialog:
 - w32, x11

gfx/display:
 - display modes
 - fullscreen display

gfx/bitmap
gfx/primitive
gfx/transform


gfx/classic:
 - backed by sw, dx9, ogl, etc.
 
gfx/accelerated:
 - superset of classic
 - backed by dx9, dx11, ogl, etc.
 - submodules: 
   - gfx/3d/shader
   - gfx/3d/texture
   - gfx/3d/pipeline
   - gfx/3d/framebuffer

gfx/modern:
 - not superset of accelerated (can't be interface-compatible)
 - low priority
 - backed by vk, dx12

io/keyboard:
 - dinput, x11

io/mouse:
 - dinput, x11
 
io/joystick:
 - dinput, x11
